{\sffamily\color{grisfonce} Cours}

\vspace{10pt}

\begin{objectifsbox}
\begin{itemize}[leftmargin=1.2em, itemsep=2pt, topsep=2pt]
  \item Reconnaître qu'une suite est convergente à partir de sa définition.
  \item Déterminer la limite d'une suite simple par le calcul ou par comparaison.
  \item Appliquer les théorèmes de comparaison pour lever une indétermination.
\end{itemize}
\end{objectifsbox}

\vspace{8pt}

\begin{definitionbox}{Limite finie d'une suite}
On dit que la suite $(u_n)$ \textbf{converge} vers un réel $\ell$ si tout intervalle ouvert contenant $\ell$ contient tous les termes de la suite à partir d'un certain rang.\\[4pt]
On note alors $\displaystyle\lim_{n \to +\infty} u_n = \ell$.
\end{definitionbox}

\vspace{6pt}

\begin{theoremebox}{Convergence monotone}
Si une suite $(u_n)$ est \textbf{croissante} et \textbf{majorée}, alors elle converge.\\[4pt]
De même, une suite \textbf{décroissante} et \textbf{minorée} converge.
\end{theoremebox}

\vspace{6pt}

\begin{methodebox}{Étudier le sens de variation}
On calcule la différence $u_{n+1} - u_n$ et on étudie son signe.\\[4pt]
Si $u_{n+1} - u_n \geqslant 0$ pour tout $n$, la suite est croissante ; si $u_{n+1} - u_n \leqslant 0$, elle est décroissante.
\end{methodebox}

\vspace{6pt}

\begin{attentionbox}{Piège classique}
Une suite \textbf{majorée} n'est pas forcément \textbf{convergente} : il faut aussi la monotonie ! Une suite peut être bornée et diverger si elle oscille indéfiniment.
\end{attentionbox}

\vspace{10pt}

{\titlefont\bfseries\color{bleupetrole} Exercices d'application}
\vspace{4pt}

\exercice{1}{\etoiles{1}\ \ On considère la suite définie par $u_0 = 1$ et $u_{n+1} = \dfrac{1}{2}u_n + 3$ pour tout $n \in \mathbb{N}$.\\
a) Montrer que la suite $(u_n)$ est croissante.\\
b) Montrer que $(u_n)$ est majorée par $6$.\\
c) En déduire que $(u_n)$ converge et déterminer sa limite.}
